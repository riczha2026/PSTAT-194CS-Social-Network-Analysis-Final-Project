\begin{frame}{Step 6 - Adjacency Matrix}
\protect\phantomsection\label{step-6---adjacency-matrix}
\begin{itemize}
\tightlist
\item
  200x200 matrix where each cell = 1 if a hyperlink exists between two

\item
  Goal: reorder nodes to reveal hidden patterns
\end{itemize}
\end{frame}

\begin{frame}{Adjacency Matrix: Reordered by Community}
\protect\phantomsection\label{adjacency-matrix-reordered-by-community}
\begin{itemize}
\tightlist
\item
  Nodes reordered by Louvain community membership
\item
  Diagonal blocks emerge --- pages in the same community densely
  hyperlink each other
\item
  Subgraph 4 (1,608 edges) shows the clearest block structure
\item
  Off-diagonal sparsity confirms pages from different communities rarely
  link directly
\end{itemize}
\end{frame}
